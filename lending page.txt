        // Intersection Observer for animations
        const observerOptions = {
            threshold: 0.1,
            rootMargin: '0px 0px -50px 0px'
        };
        
        const observer = new IntersectionObserver((entries) => {
            entries.forEach(entry => {
                if (entry.isIntersecting) {
                    entry.target.classList.add('visible');
                    
                    // Animate module line if it's the module section
                    if (entry.target.querySelector('.module-line')) {
                        document.getElementById('moduleLine').classList.add('visible');
                    }
                }
            });
        }, observerOptions);
        
        document.querySelectorAll('.fade-up').forEach(el => observer.observe(el));
        
        // Smooth scroll
        function scrollToSection(id) {
            document.getElementById(id).scrollIntoView({ behavior: 'smooth' });
        }
        
        // Toggle modules
        function toggleModules() {
            const hidden = document.getElementById('hiddenModules');
            const btn = document.getElementById('showAllModules');
            
            if (hidden.classList.contains('hidden')) {
                hidden.classList.remove('hidden');
                btn.innerHTML = '<span>Свернуть</span><i data-lucide="chevron-up" class="w-5 h-5"></i>';
                lucide.createIcons();
            } else {
                hidden.classList.add('hidden');
                btn.innerHTML = '<span>Показать все 9 модулей</span><i data-lucide="chevron-down" class="w-5 h-5"></i>';
                lucide.createIcons();
            }
        }
        
        // Form submission
        function handleSubmit(e) {
            e.preventDefault();
            const btn = e.target.querySelector('button');
            const originalText = btn.innerHTML;
            
            btn.innerHTML = '<span>Отправка...</span>';
            btn.disabled = true;
            
            setTimeout(() => {
                btn.innerHTML = '<span>Заявка отправлена! ✓</span>';
                btn.classList.remove('btn-primary');
                btn.style.background = '#22c55e';
                
                setTimeout(() => {
                    btn.innerHTML = originalText;
                    btn.classList.add('btn-primary');
                    btn.style.background = '';
                    btn.disabled = false;
                    e.target.reset();
                }, 3000);
            }, 1500);
        }
        
        // Parallax effect for hero
        window.addEventListener('scroll', () => {
            const scrolled = window.pageYOffset;
